  <script crossorigin src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
  <script crossorigin src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>

<body class="text-gray-800 antialiased">
  <div id="root"></div>

  <script type="text/babel">
    const { useState, useEffect, useMemo } = React;

    // ⚠️ 1. 請填入您的 GAS Web App 網址
    const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbzf8kJ6Ka8yGabg--MCRJ8eyucBbsGRDbceGEeH-CQDLqOMXhTCysZVrPKL0MLpSg4L/exec';
    
    // ⚠️ 2. 請填入您在 LINE Developers 申請到的 LIFF ID
    const LIFF_ID = '2009807397-WPVPBokl';

    // ==========================================
    // 🚀 核心優化：共用的 API 呼叫函式 (解決 CORS 問題)
    // ==========================================
    async function callGasApi(payload) {
      try {
        const response = await fetch(SCRIPT_URL, {
          method: 'POST',
          // 重要：使用 text/plain 避開 CORS 預檢請求，確保能正確讀取 GAS 回傳的 JSON
          headers: { 'Content-Type': 'text/plain;charset=utf-8' },
          body: JSON.stringify(payload)
        });

        if (!response.ok) throw new Error(`網路連線異常 (狀態碼: ${response.status})`);
        
        const result = await response.json();
        if (result.status === 'error') throw new Error(result.message || '後端處理發生錯誤');
        
        return result;
      } catch (err) {
        console.error('API 呼叫失敗:', err);
        throw err; // 往上拋出錯誤交由具體業務邏輯處理
      }
    }

    const productMapping = {
      1: "蕃茄 (小/喜糖)", 2: "蕃茄蜜餞 (小/喜糖)", 3: "鳥梨 (小/喜糖)", 4: "蕃茄+鳥梨 (小/喜糖)",
      5: "承租掃帚", 6: "蕃茄 (經典)", 7: "蕃茄蜜餞 (經典)", 8: "鳥梨 (經典)"
    };

    function AdminApp() {
      // 登入狀態：checking(檢查中), unauth(未登入), unauthorized_user(無權限), logged_in(已登入)
      const [authStatus, setAuthStatus] = useState('checking');
      const [userProfile, setUserProfile] = useState(null);

      // 導覽頁籤狀態 (新增 'settings')
      const [activeTab, setActiveTab] = useState('dashboard');
      
      const [orders, setOrders] = useState([]);
      const [isLoading, setIsLoading] = useState(false);
      const [alertMsg, setAlertMsg] = useState(null);
      
      const [selectedDate, setSelectedDate] = useState(new Date().toISOString().split('T')[0]);
      const [searchTerm, setSearchTerm] = useState('');
      
      // 系統設定狀態
      const [settings, setSettings] = useState({ reminderEnabled: true, reminderTime: '11:00' });
      const [isSavingSettings, setIsSavingSettings] = useState(false);

      // Modal 狀態管理
      const [resendModal, setResendModal] = useState({ isOpen: false, order: null, email: '' });
      const [isResending, setIsResending] = useState(false);

      const [editModal, setEditModal] = useState({ isOpen: false, order: null, eventDate: '', eventTime: '' });
      const [isUpdating, setIsUpdating] = useState(false);

      // UI 狀態：手機版側邊欄開關
      const [isMobileMenuOpen, setIsMobileMenuOpen] = useState(false);

      // ==========================================
      // 🔐 LINE LIFF 初始化與登入驗證
      // ==========================================
      useEffect(() => {
        async function initializeLiff() {
          try {
            await liff.init({ liffId: LIFF_ID });
            
            if (liff.isLoggedIn()) {
              const profile = await liff.getProfile();
              const res = await fetch(`${SCRIPT_URL}?action=verify_admin&userId=${profile.userId}`);
              const data = await res.json();
              
              if (data.status === 'success' && data.isAdmin) {
                setUserProfile(profile);
                setAuthStatus('logged_in');
                fetchOrders(); 
                fetchSettings(); // 取得系統設定
              } else {
                setAuthStatus('unauthorized_user'); 
              }
            } else {
              setAuthStatus('unauth'); 
            }
          } catch (err) {
            console.error('LIFF 初始化失敗', err);
            setAlertMsg("系統啟動失敗，請確認 LIFF ID 是否設定正確。(" + err.message + ")");
            setAuthStatus('unauth');
          }
        }
        
        if (!LIFF_ID || LIFF_ID === '在此填入您的_LIFF_ID') {
           setAlertMsg("請先在程式碼中填入您的 LIFF ID 才能啟用系統。");
           setAuthStatus('unauth');
        } else {
           initializeLiff();
        }
      }, []);

      const handleLogin = () => { 
        if (!LIFF_ID || LIFF_ID === '在此填入您的_LIFF_ID') {
          setAlertMsg("⚠️ 無法登入：請先填入您的 LIFF ID。");
          return;
        }
        if (window.self !== window.top) {
          setAlertMsg("⚠️ 系統安全限制：\n您目前正在預覽環境中。LINE 登入的跳轉機制會被封鎖！\n\n請將此網頁部署後開啟。");
          return;
        }
        try {
          if (!liff.isLoggedIn()) liff.login({ redirectUri: window.location.href });
        } catch (err) {
          setAlertMsg("登入功能發生錯誤 (" + err.message + ")");
        }
      };
      
      const handleLogout = () => { 
        if (liff.isLoggedIn()) { liff.logout();
        window.location.reload(); } 
      };

      // ==========================================
      // 📦 資料處理邏輯 (訂單 & 設定)
      // ==========================================
      const fetchOrders = async () => {
        setIsLoading(true);
        try {
          const res = await fetch(`${SCRIPT_URL}?action=get_all_orders`);
          const data = await res.json();
          if (data.status === 'success') setOrders(data.data);
          else setAlertMsg("資料讀取失敗：" + data.message);
        } catch (err) { 
          setAlertMsg("網路連線錯誤，請稍後再試。");
        } finally { 
          setIsLoading(false); 
        }
      };

      const fetchSettings = async () => {
        try {
          const res = await fetch(`${SCRIPT_URL}?action=get_settings`);
          const data = await res.json();
          if (data.status === 'success') {
            setSettings({ reminderEnabled: data.data.reminderEnabled, reminderTime: data.data.reminderTime });
          }
        } catch (err) { 
          console.error("設定讀取失敗", err);
        }
      };

      // 儲存系統設定 (升級採用 callGasApi)
      const handleSaveSettings = async () => {
        setIsSavingSettings(true);
        try {
          await callGasApi({ 
            action: 'save_settings', 
            reminderEnabled: settings.reminderEnabled, 
            reminderTime: settings.reminderTime 
          });
          setAlertMsg("✅ 系統設定已成功儲存並更新排程！");
        } catch (err) {
          setAlertMsg("❌ 設定儲存失敗：" + err.message);
        } finally {
          setIsSavingSettings(false);
        }
      };

      const urgentOrders = useMemo(() => {
        const today = new Date(); today.setHours(0, 0, 0, 0);
        const threeDaysLater = new Date(today); threeDaysLater.setDate(today.getDate() + 3);
        return orders.filter(o => {
          if (!o.eventDate) return false;
          const eDate = new Date(o.eventDate); return eDate >= today && eDate <= threeDaysLater;
        }).sort((a, b) => new Date(a.eventDate) - new Date(b.eventDate) || a.eventTime.localeCompare(b.eventTime));
      }, [orders]);

      const { dailyOrders, dailyMaterials } = useMemo(() => {
        const dOrders = orders.filter(o => o.eventDate === selectedDate).sort((a, b) => a.eventTime.localeCompare(b.eventTime));
        const materials = {}; let totalCandies = 0;
        dOrders.forEach(order => {
          if (!order.cart) return;
          Object.entries(order.cart).forEach(([id, qty]) => {
            if (qty <= 0) return;
            const pid = parseInt(id);
            if (!materials[pid]) materials[pid] = { name: productMapping[pid] || `未知商品(${pid})`, qty: 0 };
            materials[pid].qty += qty; if (pid !== 5) totalCandies += qty; 
          });
        });
        return { dailyOrders: dOrders, dailyMaterials: { items: materials, totalCandies } };
      }, [orders, selectedDate]);

      const filteredOrders = useMemo(() => {
        if (!searchTerm) return orders;
        const lowerTerm = searchTerm.toLowerCase();
        return orders.filter(o => 
          (o.orderNumber && o.orderNumber.toLowerCase().includes(lowerTerm)) || (o.ordererName && o.ordererName.toLowerCase().includes(lowerTerm)) || (o.ordererPhone && o.ordererPhone.includes(lowerTerm))
        );
      }, [orders, searchTerm]);

      // 發送補發 PDF 請求 (升級採用 callGasApi)
      const handleResendPDF = async () => {
        if (!resendModal.email) { 
          alert("請輸入 Email");
          return; 
        }
        setIsResending(true);
        try {
          await callGasApi({ 
            action: 'admin_resend_pdf', 
            orderNumber: resendModal.order.orderNumber, 
            email: resendModal.email 
          });
          setAlertMsg(`✅ 已成功補發訂單明細至 ${resendModal.email}`);
          setResendModal({ isOpen: false, order: null, email: '' });
        } catch (err) { 
          setAlertMsg("❌ 發送失敗：" + err.message);
        } finally { 
          setIsResending(false); 
        }
      };

      // 發送更改活動日期與時間請求 (升級採用 callGasApi)
      const handleUpdateOrderTime = async () => {
        if (!editModal.eventDate || !editModal.eventTime) { 
          alert("請選擇新的日期與時間");
          return; 
        }
        setIsUpdating(true);
        try {
          const result = await callGasApi({ 
            action: 'update_order_time', 
            orderNumber: editModal.order.orderNumber, 
            newDate: editModal.eventDate, 
            newTime: editModal.eventTime 
          });
          
          setAlertMsg(`✅ 改期成功！\n\n新訂單 PDF 已於後台重新生成，系統將自動為您刷新訂單列表。`);
          setEditModal({ isOpen: false, order: null, eventDate: '', eventTime: '' });
          
          // 等待刷新列表，確保畫面顯示的是最新的時間與資料
          await fetchOrders();
        } catch (err) { 
          setAlertMsg("❌ 修改指令發送失敗：" + err.message); 
        } finally { 
          setIsUpdating(false);
        }
      };

      // ==========================================
      // 🎨 UI 畫面渲染
      // ==========================================

      if (authStatus === 'checking') {
        return <div className="min-h-screen flex flex-col items-center justify-center bg-gray-50"><div className="w-12 h-12 border-4 border-brandRed/30 border-t-brandRed rounded-full animate-spin mb-4"></div><p className="text-gray-500 font-bold tracking-widest animate-pulse">系統安全驗證中...</p></div>;
      }

      if (authStatus === 'unauth') {
        return (
          <div className="min-h-screen flex items-center justify-center bg-gray-100 relative overflow-hidden p-6">
            <div className="absolute top-[-10%] left-[-10%] w-[50vw] h-[50vw] bg-brandWood/30 blur-[80px] rounded-full"></div>
            <div className="absolute bottom-[-10%] right-[-10%] w-[50vw] h-[50vw] bg-brandRed/10 blur-[80px] rounded-full"></div>
            <div className="bg-white rounded-3xl p-10 max-w-md w-full shadow-2xl relative z-10 text-center border border-white/50 backdrop-blur-xl">
              <div className="w-20 h-20 bg-brandRed/10 rounded-full flex items-center justify-center mx-auto mb-6 text-brandRed">
                <svg className="w-10 h-10" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M12 15v2m-6 4h12a2 2 0 002-2v-6a2 2 0 00-2-2H6a2 2 0 00-2 2v6a2 2 0 002 2zm10-10V7a4 4 0 00-8 0v4h8z"></path></svg>
              </div>
             
              <h1 className="text-2xl font-bold text-gray-800 mb-2">李伯伯糖葫蘆</h1>
              <p className="text-gray-500 mb-8 text-sm">內部管理員專屬後台</p>
              
              <button onClick={handleLogin} className="w-full bg-lineGreen hover:bg-[#05b34c] text-white font-bold py-3.5 px-4 rounded-xl shadow-md transition-transform hover:-translate-y-1 flex items-center justify-center gap-3">
                <svg viewBox="0 0 24 24" className="w-6 h-6" fill="currentColor"><path d="M24 10.304c0-5.369-5.383-9.738-12-9.738-6.616 0-12 4.369-12 9.738 0 4.814 4.269 8.846 10.036 9.608.391.084.922.258 1.057.592.122.303.079.778.039 1.085l-.171 1.027c-.053.303-.242 1.186 1.039.647 1.281-.54 6.911-4.069 9.428-6.967 1.739-1.907 2.572-3.843 2.572-5.992z"/></svg>
                使用 LINE 安全登入
              </button>
            </div>
            
            {alertMsg && (
              <div className="fixed inset-0 z-[60] flex items-center justify-center bg-black/40 backdrop-blur-sm">
                <div className="bg-white rounded-2xl p-6 max-w-sm w-full mx-4 shadow-xl">
                  <h3 className="text-lg font-bold text-gray-900 mb-3">系統訊息</h3>
                  <p className="text-gray-600 mb-6 whitespace-pre-line">{alertMsg}</p>
                  <button onClick={() => setAlertMsg(null)} className="w-full bg-brandRed text-white font-bold py-2.5 rounded-xl hover:bg-red-800">確定</button>
                </div>
              </div>
            )}
          </div>
        );
      }

      if (authStatus === 'unauthorized_user') {
        return (
          <div className="min-h-screen flex flex-col items-center justify-center bg-gray-50 p-6">
            <div className="bg-red-50 border border-red-200 text-brandRed rounded-3xl p-10 max-w-md w-full shadow-lg text-center">
              <svg className="w-16 h-16 mx-auto mb-4" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M12 9v2m0 4h.01m-6.938 4h13.856c1.54 0 2.502-1.667 1.732-3L13.732 4c-.77-1.333-2.694-1.333-3.464 0L3.34 16c-.77 1.333.192 3 1.732 3z"></path></svg>
              <h2 className="text-xl font-bold mb-2">權限不足</h2>
              <p className="text-red-600/80 mb-8 text-sm">抱歉，您的 LINE 帳號不在後台管理員名單中，無法存取此系統資料。</p>
              <button onClick={handleLogout} className="bg-white text-brandRed border border-brandRed/30 px-6 py-2 rounded-xl font-bold hover:bg-red-50 transition-colors">切換帳號或登出</button>
            </div>
          </div>
        );
      }

      return (
        <div className="flex h-screen overflow-hidden bg-gray-100 relative">
          
          <div className="md:hidden absolute top-0 left-0 w-full h-16 bg-white shadow-sm z-30 flex items-center justify-between px-4 border-b border-gray-200">
            <h1 className="text-lg font-bold text-brandRed tracking-widest">李伯伯糖葫蘆</h1>
            <button onClick={() => setIsMobileMenuOpen(true)} className="text-gray-600 p-2 focus:outline-none">
              <svg className="w-6 h-6" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M4 6h16M4 12h16M4 18h16"></path></svg>
            </button>
          </div>

          {isMobileMenuOpen && (
            <div className="md:hidden fixed inset-0 bg-black/50 z-40 transition-opacity" onClick={() => setIsMobileMenuOpen(false)}></div>
          )}

          {/* 左側導覽列 */}
          <aside className={`fixed md:static inset-y-0 left-0 w-64 bg-white shadow-2xl md:shadow-md flex flex-col z-50 shrink-0 transform transition-transform duration-300 ease-in-out ${isMobileMenuOpen ? 'translate-x-0' : '-translate-x-full md:translate-x-0'}`}>
            <div className="p-6 border-b border-gray-100 flex justify-between items-center md:block text-center md:text-center">
              <div className="text-left md:text-center">
                <h1 className="text-xl font-bold text-brandRed tracking-widest">李伯伯糖葫蘆</h1>
                <p className="text-xs text-gray-400 mt-1">後台管理系統 v1.1</p>
              </div>
              <button onClick={() => setIsMobileMenuOpen(false)} className="md:hidden text-gray-400 p-1 bg-gray-50 rounded-lg">
                <svg className="w-6 h-6" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M6 18L18 6M6 6l12 12"></path></svg>
              </button>
            </div>
            
            <div className="px-6 py-4 bg-gray-50 border-b border-gray-100 flex items-center gap-3">
              {userProfile?.pictureUrl ? <img src={userProfile.pictureUrl} className="w-10 h-10 rounded-full border-2 border-white shadow-sm"/> : <div className="w-10 h-10 rounded-full bg-brandRed/10 flex items-center justify-center text-brandRed font-bold">{userProfile?.displayName?.charAt(0)}</div>}
              <div className="overflow-hidden">
                <p className="text-sm font-bold text-gray-800 truncate">{userProfile?.displayName}</p>
                <p className="text-[10px] text-green-600 font-bold">● 已連線</p>
              </div>
            </div>

            <nav className="flex-1 p-4 space-y-2">
              <button onClick={() => { setActiveTab('dashboard'); setIsMobileMenuOpen(false); }} className={`w-full flex items-center gap-3 px-4 py-3 rounded-xl transition-colors ${activeTab === 'dashboard' ? 'bg-brandRed/10 text-brandRed font-bold' : 'text-gray-600 hover:bg-gray-50'}`}>
                <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M8 7V3m8 4V3m-9 8h10M5 21h14a2 2 0 002-2V7a2 2 0 00-2-2H5a2 2 0 00-2 2v12a2 2 0 002 2z"></path></svg>
                排程與備料
              </button>
              <button onClick={() => { setActiveTab('orders'); setIsMobileMenuOpen(false); }} className={`w-full flex items-center gap-3 px-4 py-3 rounded-xl transition-colors ${activeTab === 'orders' ? 'bg-brandRed/10 text-brandRed font-bold' : 'text-gray-600 hover:bg-gray-50'}`}>
                <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M9 5H7a2 2 0 00-2 2v12a2 2 0 002 2h10a2 2 0 002-2V7a2 2 0 00-2-2h-2M9 5a2 2 0 002 2h2a2 2 0 002-2M9 5a2 2 0 012-2h2a2 2 0 012 2m-6 9l2 2 4-4"></path></svg>
                訂單總覽
              </button>
              {/* 新增系統設定頁籤 */}
              <button onClick={() => { setActiveTab('settings'); setIsMobileMenuOpen(false); }} className={`w-full flex items-center gap-3 px-4 py-3 rounded-xl transition-colors ${activeTab === 'settings' ? 'bg-brandRed/10 text-brandRed font-bold' : 'text-gray-600 hover:bg-gray-50'}`}>
                <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M10.325 4.317c.426-1.756 2.924-1.756 3.35 0a1.724 1.724 0 002.573 1.066c1.543-.94 3.31.826 2.37 2.37a1.724 1.724 0 001.065 2.572c1.756.426 1.756 2.924 0 3.35a1.724 1.724 0 00-1.066 2.573c.94 1.543-.826 3.31-2.37 2.37a1.724 1.724 0 00-2.572 1.065c-.426 1.756-2.924 1.756-3.35 0a1.724 1.724 0 00-2.573-1.066c-1.543.94-3.31-.826-2.37-2.37a1.724 1.724 0 00-1.065-2.572c-1.756-.426-1.756-2.924 0-3.35a1.724 1.724 0 001.066-2.573c-.94-1.543.826-3.31 2.37-2.37.996.608 2.296.07 2.572-1.065z"></path><path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M15 12a3 3 0 11-6 0 3 3 0 016 0z"></path></svg>
                系統設定
              </button>
            </nav>
            <div className="p-4 border-t border-gray-100 space-y-2">
              <button onClick={() => { fetchOrders(); fetchSettings(); setIsMobileMenuOpen(false); }} className="w-full flex items-center justify-center gap-2 text-sm text-gray-500 hover:text-gray-800 transition-colors py-2">
                <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M4 4v5h.582m15.356 2A8.001 8.001 0 004.582 9m0 0H9m11 11v-5h-.581m0 0a8.003 8.003 0 01-15.357-2m15.357 2H15"></path></svg> 重新載入
              </button>
              <button onClick={handleLogout} className="w-full flex items-center justify-center gap-2 text-sm text-red-500 hover:bg-red-50 transition-colors py-2 rounded-lg">
                <svg className="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M17 16l4-4m0 0l-4-4m4 4H7m6 4v1a3 3 0 01-3 3H6a3 3 0 01-3-3V7a3 3 0 013-3h4a3 3 0 013 3v1"></path></svg> 安全登出
              </button>
            </div>
          </aside>

          {/* 右側主內容區 */}
          <main className="flex-1 overflow-y-auto p-4 pt-20 md:p-8 md:pt-8 relative scrollbar-hide">
            
            {isLoading && (
               <div className="absolute inset-0 bg-white/50 backdrop-blur-[2px] z-20 flex items-center justify-center">

                  <div className="mb-6">
                    <label className="block text-sm font-bold text-gray-700 mb-1">接收人 Email</label>
                    <input type="email" value={resendModal.email} onChange={e => setResendModal({...resendModal, email: e.target.value})} className="w-full px-4 py-3 bg-gray-50 border border-gray-200 rounded-xl focus:ring-2 focus:ring-brandRed outline-none" placeholder="請輸入電子信箱" />
                  </div>
                  
                  <div className="flex gap-3">
                    <button onClick={() => setResendModal({ isOpen: false, order: null, email: '' })} className="flex-1 px-4 py-2.5 bg-gray-100 text-gray-600 font-bold rounded-xl hover:bg-gray-200">取消</button>
                    <button onClick={handleResendPDF} disabled={isResending} className="flex-1 px-4 py-2.5 bg-brandRed text-white font-bold rounded-xl hover:bg-red-800 disabled:opacity-50">
                      {isResending ? '發送中...' : '確認發送'}
                    </button>
                  </div>
                </div>
              </div>
            )}

          </main>
        </div>
      );
    }

    const root = ReactDOM.createRoot(document.getElementById('root'));
    root.render(<AdminApp />);
  </script>
